1function [xi, yi] =  my_intersection(x,y)
 2dx = diff(x);  %# Take the differences down each column
 3dy = diff(y);
 4den = dx(1)*dy(2)-dy(1)*dx(2);  %# Precompute the denominator
 5if(den == 0)
 6    error('Parallella linjer');
 7end
 8ua = (dx(2)*(y(1)-y(3))-dy(2)*(x(1)-x(3)))/den;
 9ub = (dx(1)*(y(1)-y(3))-dy(1)*(x(1)-x(3)))/den;
10
11xi = x(1)+ua*dx(1);

13end
